```cpp
class Solution {
public:
    TreeNode* pre = nullptr;
    TreeNode* res = nullptr;
    TreeNode* increasingBST(TreeNode* root) {
        if(root == nullptr) return nullptr;
        increasingBST(root->left);

        if(res == nullptr) res = root;
        root->left = nullptr;
        if(pre != nullptr){
            pre->right = root;
        }

        pre = root;
        increasingBST(root->right);
        return res;
    }
};
```
