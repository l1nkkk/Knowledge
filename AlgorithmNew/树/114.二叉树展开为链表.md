class Solution {
public:
    // 定义：将以root为根的树拉平
    void flatten(TreeNode* root) {
        if(root == nullptr) return;
        // 1.将左右子树拉平
        flatten(root->left);
        flatten(root->right);

        /**** 后序遍历位置 ****/
        // 2.拉平后，将右边加到左边的尾部
        if(root->left != nullptr){
            TreeNode* lend = root->left;
            while(lend->right != nullptr){
                lend = lend->right;
            }
            lend->right = root->right;
            root->right = root->left;
            root->left = nullptr;
        }

        return;
    }
};
