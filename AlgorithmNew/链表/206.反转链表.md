> 题目
- 反转整个链表

> 代码

```
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head == nullptr) return nullptr;
        if(head->next == nullptr){
            pre =  head;
            return head;
        }
        ListNode* rtn = reverseList(head->next);
        pre->next = head;
        pre = head;
        head->next = nullptr;
        return rtn;
    }
private:
    ListNode* pre;
};
```
