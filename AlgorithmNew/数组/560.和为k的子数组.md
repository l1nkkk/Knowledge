```cpp
class Solution {
public:
int subarraySum(vector<int>& nums, int k) {
    // preSum ==> times
    unordered_map<int, int> recordPreSum;
    recordPreSum[0] = 1;
    int preSum = 0;
    int res = 0;
    for(auto n : nums){
        preSum += n;
        if(recordPreSum.count(preSum-k))
            res += recordPreSum[preSum-k];

        ++recordPreSum[preSum];
    }
    return res;
}
};
```
