- 中序遍历
> 题目


输入一棵二叉搜索树，将该二叉搜索树转换成一个排序的**循环双向链表**。要求不能创建任何新的节点，只能调整树中节点指针的指向。
- https://leetcode-cn.com/problems/er-cha-sou-suo-shu-yu-shuang-xiang-lian-biao-lcof/

> 代码
```cpp


class Solution {
public:
    Node* head = nullptr;
    Node* pre = nullptr;
    Node* treeToDoublyList(Node* root) {
        if(root == nullptr) return nullptr;
        dfs(root);
        head->left = pre;
        pre->right = head;
        return head;
    }
    void dfs(Node* root){
        if(root == nullptr) return;
        dfs(root->left);
        if(head == nullptr){
            head = root;
        }
        if(pre != nullptr){
            pre->right = root;
        }
        root->left = pre;
        pre = root;
        dfs(root->right);
    }
};
```
