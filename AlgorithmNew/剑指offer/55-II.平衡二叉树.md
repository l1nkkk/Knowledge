class Solution {
public:
    bool isBalanced(TreeNode* root) {
        return dfs(root) == -1 ? false : true;
    }
    int dfs(TreeNode *root){
        if(root == nullptr) return 0;
        auto l = dfs(root->left);
        if(l == -1) return -1;
        auto r = dfs(root->right);
        if(r == -1) return -1;
        if(abs(l-r) > 1) return -1;
        return max(l,r) +1;
    }
};
