class Solution {
public:
    int singleNumber(vector<int>& nums) {
        unordered_map<int,int> record;
        for(auto a: nums){
            ++record[a];
        }
        for(auto a : record){
            if(a.second == 1)
                return a.first;
        }
        return 0;
    }
};
