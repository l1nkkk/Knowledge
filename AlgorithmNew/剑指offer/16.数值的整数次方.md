class Solution {
public:
    double myPow(double x, int n) {
        double res = 1;
        long b = n;
        bool flag = n < 0 ? true : false;
        if(flag) b = -b;
        while(b){
            if(b & 1){
                res *= x;
            }
            x*= x;
            b >>= 1;
        }
        return flag? 1/res : res;
    }
};
