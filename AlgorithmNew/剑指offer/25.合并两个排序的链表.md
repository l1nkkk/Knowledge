class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode* dumy = new ListNode(0);
        ListNode* it = dumy;
        while(l1 != NULL && l2 != NULL){
            if( l1->val > l2->val){
                it->next = l2;
                l2 = l2->next;
            }
            else{
                it->next = l1;
                l1 = l1->next;
            }
                
            it = it->next;
        }
        if(l1 != NULL) it->next = l1;
        if(l2 != NULL) it->next = l2;

        return dumy->next;
    }
};
