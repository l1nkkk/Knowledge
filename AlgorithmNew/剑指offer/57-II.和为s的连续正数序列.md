```cpp
class Solution {
public:
    vector<vector<int>> findContinuousSequence(int target) {
        int left=1, right=1;
        vector<vector<int>> res;
        while(right < target){
            int cal = ((left+right)*(right-left+1))/2;
            if(cal == target){
                vector<int> temp;
                for(int i = left; i <= right; ++i){
                    temp.push_back(i);
                }
                res.push_back(temp);
                ++left;
            }else if(cal < target){
                ++right;
            }else if(cal > target){
                ++left;
            }
        }
        return res;
    }
};
```
