```cpp
class Solution {
public:
    TreeNode* mirrorTree(TreeNode* root) {
            if(root == nullptr) return nullptr;
            auto lp = mirrorTree(root->left);
            auto rp = mirrorTree(root->right);

            root->left = rp;
            root->right = lp;
            return root;
    }
};
```
