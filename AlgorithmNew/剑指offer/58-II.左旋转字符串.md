```cpp
class Solution {
public:
    string reverseLeftWords(string s, int n) {
        reverse(s.begin(), s.end());
        reverse(s.begin(), s.begin()+s.size()-n);
        reverse(s.begin()+s.size()-n, s.end());
        return s;
    }
};
```
