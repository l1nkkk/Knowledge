> 难度：简单
- 中序遍历有序的性质

> 题目
- 给定一棵二叉搜索树，请找出其中第k大的节点。
> 代码

```cpp
class Solution {
public:
    int kthLargest(TreeNode* root, int k) {
        dfs(root,k);
        return res;

    }
    int curK = 0;
    int res = 0;
    void dfs(TreeNode* root, int k){
        if(root == NULL) return;
        if(curK == k) return;
        dfs(root->right, k);
        ++curK;
        if(curK == k){
            res = root->val;
            return;
        }
        dfs(root->left, k);
    }
};
```
