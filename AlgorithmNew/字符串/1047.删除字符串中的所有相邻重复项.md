class Solution {
public:
    string removeDuplicates(string s) {
        string res = s.substr(0,1);
        for(int i = 1; i < s.size(); ++i){
            if(!res.empty() && res.back() == s[i]){
                res.pop_back();
            }else
                res.push_back(s[i]);
        }
        return res;
    }
};
