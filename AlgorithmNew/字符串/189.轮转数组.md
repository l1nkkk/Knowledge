class Solution {
public:
//    [lo,hi)
    void reverse(vector<int>& nums, int lo, int hi){
        while(lo+1 < hi){
            nums[lo] ^= nums[hi-1];
            nums[hi-1] ^= nums[lo];
            nums[lo] ^= nums[hi-1];
            lo++;
            hi--;
        }
    }
    void rotate(vector<int>& nums, int k) {
        reverse(nums,0, nums.size());
        k = k %nums.size();

        reverse(nums,0,k);
        reverse(nums, k, nums.size());
    }
};
