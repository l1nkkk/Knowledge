```cpp
class Solution {
public:
    int mySqrt1(int x) {
        if(x == 0) return 0;
        uint64_t res =  exp(0.5* log(x));
        return (res+1)*(res+1) <= x? res+1 : res;
    }
    int mySqrt(int x) {
        if(x == 0 || x== 1)return x;
        int left,right;
        left = 0;
        right = x;
        // [left,right]，也不一定要区间为空我才退出，看具体的题目，要会变通
        while(left < right){
            uint64_t mid = left + (right - left)/2;
            uint64_t rtn=mid*mid;
            if(rtn > x){
                right = mid;
            }else
                left = mid+1;
        }
        // 重点：为什么要-1，因为只要动过right,right*right>x。最后退出时left == right
        return left-1;
    }
};

```
